<Step>Install the following dependencies</Step>

<NpmCommand value='npm install clsx tailwind-merge @radix-ui/react-label' />

<Step>Add a classname utility function</Step>

```ts
import { clsx, type ClassValue } from 'clsx';
import { twMerge } from 'tailwind-merge';

export const cn = (...inputs: ClassValue[]) => {
  return twMerge(clsx(inputs));
};
```

## Usage

```ts
import { FloatingInput } from '@/components/ui/floating-label-input';

export default function FloatingInputDemo() {
  return (
    <FloatingInput id='floating-label-input-demo' label='floating input' />
  );
}
```
